# Group Anagrams

```py
class Solution:
    def groupAnagrams(self, strs):
        """
        :type strs: List[str]
        :rtype: List[List[str]]
        """
        # 键可以用tuple(sorted(item))
        
        dic={}
        for item in strs:
            tmp=''.join(sorted(item))
            if tmp in dic:
                #dic[tmp].append(item)
                dic.get(tmp).append(item)
            else:
                dic[tmp]=[item]
        #Res=[]
        #for i in dic.keys():
            #Res.append(dic.get(i))
        #return Res
        return list(dic.values())
```
