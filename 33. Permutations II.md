class Solution:
    def permuteUnique(self, nums):
        """
        :type nums: List[int]
        :rtype: List[List[int]]
        """
        
        def recu(nums,res):
            if len(nums)==0:
                Res.append(res)
                return
            for index,item in enumerate(list(set(nums))):
                tmp=nums.copy()
                tmp.remove(item)
                recu(tmp,res+[item])
        
        if len(nums)<2:
            return [nums]
        
        Res=[]
        recu(nums,[])
        return Res
