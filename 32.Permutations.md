# Permutations

```py
class Solution:
    def permute(self, nums):
        """
        :type nums: List[int]
        :rtype: List[List[int]]
        """
        def recur(nums,res):
            if len(nums)==0:
                Res.append(res)
                return
            for index,item in enumerate(nums):            
                recur(nums[0:index]+nums[index+1:],res+[item])
        
        if len(nums)<2:
            return [nums]
        Res=[]
        recur(nums,[])
            
        return Res
```
